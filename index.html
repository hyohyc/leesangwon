// server.js import express from 'express'; import fetch from 'node-fetch'; import cors from 'cors'; import dotenv from 'dotenv'; dotenv.config(); const app = express(); app.use(cors()); app.use(express.json()); const PORT = process.env.PORT || 3000; app.post('/api/chat', async (req, res) => { try { const { messages, character } = req.body; // 캐릭터 정보 const systemMsg = { role: "system", content: `너는 ${character.name}, 나이:${character.age || "알 수 없음"}, 성격:${character.personality}, 연인:${character.lover || "없음"}, 짝사랑:${character.crush || "없음"}, 현재 감정:${JSON.stringify(character.mood)}. 단톡방에는 ${character.name} 외에 다른 참가자들이 있음. 이전 대화 기억. 캐릭터 성격대로 자연스럽게 답변.` }; const payload = { model: "gpt-4o-mini", messages: [systemMsg, ...messages], temperature: 0.8 }; const response = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${process.env.OPENAI_API_KEY}` }, body: JSON.stringify(payload) }); const data = await response.json(); res.json({ reply: data.choices?.[0]?.message?.content || "응답 없음" }); } catch (err) { console.error(err); res.json({ reply: "오류 발생" }); } }); app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
